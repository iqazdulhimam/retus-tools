⚡ Quick Click Bot GENERIC

Buatan sendiri — klik tombol secara cepat pas waktunya tiba. Bisa dipake di platform mana aja.

Cara pake:
Buka halaman tujuan
F12 → Console
Paste script → Enter
Ganti "TOMBOL" dengan teks tombol yg mau diklik (misal "Pesan", "Beli", "Checkout")
// ⚡ GENERIC AUTO CLICK
// Ganti "TOMBOL" dengan teks tombol tujuan

(function(){
  const TARGET = "TOMBOL"; // << GANTI INI
  let r=0,t=0,s=Date.now();

  console.log(`%c⚡ AUTO CLICK: "${TARGET}" — Aktif!`,'color:#00e5ff;font-weight:bold');

  function ck(){
    if(r)return; t++;
    const btns=document.querySelectorAll('button,a,[role="button"],input[type="submit"]');
    for(const b of btns){
      if(b.textContent.toLowerCase().includes(TARGET.toLowerCase())&&!b.disabled){
        if(b.offsetParent!==null){
          console.log(`%c✅ "${b.textContent.trim()}" DIKLIK!`,'color:#00e676;font-weight:bold;font-size:14px');
          b.click();
          r=1;
          return;
        }
      }
    }
    if(!r){
      if(t%10===0) console.log(`[${((Date.now()-s)/1000).toFixed(1)}s] 🔍 Nunggu "${TARGET}"...`);
      setTimeout(ck,100);
    }
  }
  ck();
  window.stop=()=>{r=1;console.log('⏹️ Stopped')};
})();
