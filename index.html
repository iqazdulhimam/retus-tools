Auto klik tombol "Pesan" pas jam buka.

Cara:
Buka halaman event Loket.com
Pilih kategori tiket (JANGAN klik Pesan dulu)
F12 → Console → paste → Enter
(function(){
  let r=0;
  console.log('%c🎫 LOKET MONITOR','color:#00e5ff;font-weight:bold');
  function ck(){
    if(r)return;
    const b=document.querySelectorAll('button,a,div[role="button"]');
    for(const n of b){
      const t=n.textContent.toLowerCase();
      if(t.includes('pesan')||t.includes('beli')||t.includes('checkout')||t.includes('order')){
        if(!n.disabled&&n.offsetParent!==null){
          console.log(`%c✅ ${n.textContent.trim()}`,'color:#00e676;font-size:14px');
          n.click();r=1;console.log('%c🎉 BERHASIL!','color:#00e676;font-weight:bold;font-size:16px');
          return;}}}
    setTimeout(ck,100);}
  ck();console.log('🔄 Menunggu tombol...');})();
